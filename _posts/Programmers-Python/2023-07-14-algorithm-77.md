```python
def solution(board, moves):
    doll = [0]
    missing_doll = 0
    for m in moves:
        for i in range(len(board)):
            if board[i][m-1] != 0:
                if doll[-1] == board[i][m-1]:
                    doll.pop()
                    board[i][m-1] = 0
                    missing_doll += 2
                else:
                    doll.append(board[i][m-1])
                    board[i][m-1] = 0
                break
    return missing_doll
```
