# 1. 라이브러리 및 데이터 로드
```python
# 데이터프레임 형태로 만들기위해 pandas 로드
import pandas as pd
# sklean 내장 데이터셋인 보스턴 주택가격 데이터셋 로드
from sklearn.datasets import load_boston
# 보스턴 데이터셋을 boston_dataset 변수에 저장
boston_dataset = load_boston()
```
